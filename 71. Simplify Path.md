#####Problem
Given an absolute path for a file (Unix-style), simplify it.

<pre>
For example,
path = "/home/", => "/home"
path = "/a/./b/../../c/", => "/c"
</pre>

#####Solution
```python
class Solution(object):
    def simplifyPath(self, path):
        """
        :type path: str
        :rtype: str
        """
        
        folders = path.split("/")
        stack = []
        
        for folder in folders:
            if len(folder) == 0 or folder == ".":
                continue
            
            if folder == "..":
                if len(stack) != 0:
                    stack.pop()
            else:
                stack.append(folder)
        
        res = []
        for folder in stack:
            res.append("/")
            res.append(folder)
        
        return "/" if len(res) == 0 else "".join(res)
```
