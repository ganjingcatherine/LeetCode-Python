#####Problem
Given two binary strings, return their sum (also a binary string).
<pre>
For example,
a = "11"
b = "1"
Return "100".
</pre>

#####Solution
```python
class Solution(object):
    def addBinary(self, a, b):
        """
        :type a: str
        :type b: str
        :rtype: str
        """
        res = []
        carry = "0"
        
        if len(a) < len(b):
            tmp = a
            a = b
            b = tmp
        
        for i in xrange(len(a)):
            if i < len(b):
                tmpSum = [a[-1 - i], b[-1 - i], carry]
                if tmpSum.count("1") >= 2:
                    carry = "1"
                    tmpSum = "1" if tmpSum.count("1") == 3 else "0"
                else:
                    carry = "0"
                    tmpSum = "1" if tmpSum.count("1") > 0 else "0"
                res.append(tmpSum)
            else:
                tmpSum = ""
                if carry == a[-1 - i] == "1":
                    tmpSum = "0"
                    carry = "1"
                else:
                    tmpSum = "1" if carry != a[-1 - i] else "0"
                    carry = "0"
                res.append(tmpSum)
        
        if carry == "1":
            res.append(carry)
        
        res.reverse()
        return "".join(res)
```
